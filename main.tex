\newcommand{\lowerint@}[2]{%
  \begingroup
  \sbox\z@{$\m@th#1\int$}%
  \lowup@{l}{\underline}{#1}%
  \endgroup
}
\newcommand{\upperint@}[2]{%
  \begingroup
  \sbox\z@{$\m@th#1\int$}%
  \lowup@{r}{\overline}{#1}%
  \endgroup
}
\newcommand{\lowup@}[3]{%
  % #1 = l (lower) or r (upper)
  % #2 = \underline (lower) or \overline (upper)
  % #3 = math style
  \rlap{%
    \hspace{0.05\wd\z@}%
    \makebox[0.9\wd\z@][#1]{%
      $\m@th#2{%
        \hspace{0.4\wd\z@}%
        \ifx#3\displaystyle\else\hspace{0.1\wd\z@}\fi
        \vphantom{\copy\z@}%
      }$%
    }%
    \hspace{0.05\wd\z@}%
  }%
}
